FROM golang:1.24-alpine AS build
WORKDIR /app
ENV CGO_ENABLED=0
ENV GOOS=linux
ARG GIT_USER
ENV GOPRIVATE=gitlab.mycompany.com
RUN apk add --no-cache git
COPY go.mod ./
RUN --mount=type=cache,target=/root/go/pkg/mod --mount=type=secret,id=gl_token \
    GH_TOKEN=$(cat /run/secrets/gl_token) && echo "machine gitlab.mycompany.com login ${GIT_USER} password ${GH_TOKEN}" >> $HOME/.netrc && go mod download && rm -f $HOME/.netrc
COPY . .
RUN --mount=type=cache,target=/root/.cache/go-build --mount=type=cache,target=/root/go/pkg/mod \
    go build -o /app/api .

FROM alpine:latest
RUN apk --no-cache add ca-certificates tzdata
RUN addgroup -S app && adduser -S -G app app
WORKDIR /app
COPY --from=build /app/api /app/api
USER app
ENTRYPOINT ["/app/api"]
